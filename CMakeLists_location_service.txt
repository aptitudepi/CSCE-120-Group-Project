
cmake_minimum_required(VERSION 3.21)
project(LocationService)

find_package(Qt6 REQUIRED COMPONENTS Core Network HttpServer Sql)

qt_add_executable(LocationService
    LocationService.cpp
    LocationService.h
    main.cpp
)

target_link_libraries(LocationService
    Qt6::Core
    Qt6::Network
    Qt6::HttpServer
    Qt6::Sql
)

qt_finalize_executable(LocationService)
