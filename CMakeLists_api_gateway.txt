
cmake_minimum_required(VERSION 3.21)
project(ApiGatewayService)

find_package(Qt6 REQUIRED COMPONENTS Core Network HttpServer)

qt_add_executable(ApiGatewayService
    ApiGatewayService.cpp
    ApiGatewayService.h
    main.cpp
)

target_link_libraries(ApiGatewayService
    Qt6::Core
    Qt6::Network
    Qt6::HttpServer
)

qt_finalize_executable(ApiGatewayService)
