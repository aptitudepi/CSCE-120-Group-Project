
cmake_minimum_required(VERSION 3.21)
project(DatabaseService)

find_package(Qt6 REQUIRED COMPONENTS Core Network HttpServer Sql)

qt_add_executable(DatabaseService
    DatabaseService.cpp
    DatabaseService.h
    main.cpp
)

target_link_libraries(DatabaseService
    Qt6::Core
    Qt6::Network
    Qt6::HttpServer
    Qt6::Sql
)

qt_finalize_executable(DatabaseService)
