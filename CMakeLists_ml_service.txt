
cmake_minimum_required(VERSION 3.21)
project(MLService)

find_package(Qt6 REQUIRED COMPONENTS Core Network HttpServer)

qt_add_executable(MLService
    MLService.cpp
    MLService.h
    main.cpp
)

target_link_libraries(MLService
    Qt6::Core
    Qt6::Network
    Qt6::HttpServer
)

qt_finalize_executable(MLService)
