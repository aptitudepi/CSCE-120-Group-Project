
cmake_minimum_required(VERSION 3.21)
project(AlertService)

find_package(Qt6 REQUIRED COMPONENTS Core Network HttpServer Sql)

qt_add_executable(AlertService
    AlertService.cpp
    AlertService.h
    main.cpp
)

target_link_libraries(AlertService
    Qt6::Core
    Qt6::Network
    Qt6::HttpServer
    Qt6::Sql
)

qt_finalize_executable(AlertService)
